//	--> 바이너리 모드와 텍스트 모드의 차이 때문에 10바이트가 더 기록됨

using namespace std;

//파일 정수  "정수 1000개 메모리 그대로 저장" 에 있는 정수 1000개를
//읽어(read) 정렬한 후 화면에 출력하라
int main()
{
	int * p = new int[1000];
	ifstream in("정수 1000개 메모리 그대로 저장", ios::binary);
	
	in.read((char * )p, sizeof(int) * 1000);

	qsort(p, 1000, sizeof(int), [](const void * a, const void * b) {
		return *(int *)a - *(int *)b; 
	});

	for (int i = 0; i < 20; ++i)
		cout << p[i] << endl;
	save();
}
