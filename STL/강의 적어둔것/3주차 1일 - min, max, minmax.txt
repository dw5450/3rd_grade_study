#include <iostream>
#include "save.h"
#include <algorithm>
using namespace std;

class Dog {
	int n;
public:
	Dog(int n): n(n) {}
	int getN() const { return n; };
	operator int()const { return n; }
};
//min, max, minmax

int main()
{
	Dog a(10), b{ 20 };

	cout << min(a, b, [](const Dog & a, const Dog & b) {
		return a.getN() < b.getN();
	}) << endl;
	//프로그래머가 얼마든지 의미를 바꿀 수 있다

	save();
}
