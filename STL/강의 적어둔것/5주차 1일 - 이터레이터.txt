int main()
{
	vector<string> v;

	ifstream in("소스.cpp");
	string s;

	while ( in >> s)
		v.push_back(s);
	
	auto beg = v.begin();
	auto end = v.end();

	for (auto i = beg; i < end; ++i) {
		string s = *i;
		for (auto i = s.rend(); i < s.rbegin(); ++i)
			cout << *i << endl;
		cout << *i << endl;
	}
	
	cout << "__________메인문 종료 ____________" << endl;
};
