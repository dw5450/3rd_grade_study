//리스트에 삽입 삭제는 O(1)

#include <iostream>
#include <list>

using namespace std;

int main()
{
	list<int> v;

	v.push_back(1);

	auto p = v.begin();
	cout << *p << endl;

	v.push_back(2);
	cout << *p << endl;
	p++;
	//다음 명령어는 실행시 지워야 할 것 입니다.


}
