#include "save.h"
using namespace std;
int main()
{
	// 소스.cpp 파일에처 서층ㅁ 나오는 숫자를 찾아보세요

	ifstream in("소스.cpp");

	auto p = find_if(istream_iterator<char>(in), istream_iterator<char>(), [] (char c){
		return isdigit(c);
	});

	if (p != istream_iterator<char>()) {
		cout << "숫자를 찾음" << *(p)<<endl;
	}
	else {
		cout << "못 찾음" << endl;
	}

	save("흥미위주의 예제.txt");

	
}
