using namespace std;

//컨테이너를 다양한 생성자로 생성해보는 예제 - 49쪽

int main()
{
	vector<Model > vec { 1 };							//begin(vec), end(vec)			//메모리가 붙어일을 경우

	cout << "---------vec 생성 후 ------------" << endl;
	for (auto & v : vec)								// 쓸 때 없는 복사를 막아주자
		v.Show();
	cout << endl;

	cout << "---------vec2 생성 전 ------------" << endl;
	vector<Model> vec2;
	vec2 = move(vec);
	cout << "---------vec2 생성 후 ------------" << endl;
	cout << vec2.size() << endl;
	cout << vec.size() << endl;
	for (auto & v : vec2)
		v.Show();
	cout << endl;




	cout << "__________메인문 종료 ____________" << endl;

};
