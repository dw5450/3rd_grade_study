#include <iostream>
#include <algorithm>			//min,max, minmax를 사용하기 위한 헤더

int main()
{
	std::cout << std::min({ 1, -2, -3,4 ,-5,6 ,12 }) << std::endl;
	std::cout << std::min({ 1, -2, -3,4 ,-5,6 ,12 }, [](int n1, int n2)
	{return abs(n1) < abs(n2);}) << std::endl;

	std::cout << std::max({ 1, -2, -3,4 ,-5,6 ,12 }) << std::endl;
	std::cout << std::max({ 1, -2, -3,4 ,-5,6 ,12 }, [](int n1, int n2)
	{return abs(n1) < abs(n2);}) << std::endl;

	std::cout << std::minmax({ 1, -2, -3,4 ,-5,6 ,12 }).first << std::endl;
	std::cout << std::minmax({ 1, -2, -3,4 ,-5,6 ,12 }).second << std::endl;
	std::cout << std::minmax({ 1, -2, -3,4 ,-5,6 ,12 }, [](int n1, int n2)
	{return abs(n1) < abs(n2);}).first << std::endl;
	std::cout << std::minmax({ 1, -2, -3,4 ,-5,6 ,12 }, [](int n1, int n2)
	{return abs(n1) < abs(n2);}).second << std::endl;
}
