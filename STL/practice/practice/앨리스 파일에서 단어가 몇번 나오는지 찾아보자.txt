//히어로와 전투력수치를 맵으로 만들어 보자

int main()
{
	string filename = "이상한 나라의 앨리스.txt";
	//이 파일은 UTF-8 형식의 파일로 BOM(Byte order makr)바이트가 기록되어 있다.
	// 3바이트를 무시해야 진짜 텍스트를 읽을 수 있다.
	ifstream in(filename);

	if (!in) {
		cout << filename << "을 열 수 없습니다." << endl;
		return 0;
	}

	char c;
	in >> c >> c >> c;		// BOM을 무시한다.

	map<string, int> simap;

	string s;
	while ( in  >> s){
		simap[s]++;
		//cimap.operater[](c)++;
	}

	// 찾는 단어가 몇번 출현하는지 알려주세요.

	string word;
	while(true)
	{
		cout << "찾으실 단어는요?    ";

		cin >> word;

		auto itor = simap.find(word);


		if (itor != simap.end())
			cout << word << "는 " << itor->second << "번 나와요" << endl;

		else cout << word << "는 이상한 나라의 앨리스에 없어요" << endl;

		if (word == "break")
			break;
	}
	save("앨리스 파일에서 단어가 몇번 나오는지 찾아보자.txt");

}
