
#역방향 str 반환
def reverse_str(s):
    l=list(s)
    rs =""
    for i in range(len(l)):
        rs += l.pop()

    return rs

#역방향 정수 반환
def reverse_n(n):
    result = 0
    while (n > 0):
        result = result * 10 + n %10
        n = n//10

    return result

def lotto():

#리버스 구구단
def reverse_99dan(n1, n2):
    result = 0
    for j in range (n2 + 1):
        if result < reverse_n(n1 * j):
            result = reverse_n(n1 * j)

    return result
