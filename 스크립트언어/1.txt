		숙제 - 1번과 2번 해결책의 차이점을 알아보자

#include iostream

using namespace std;

class Dog
{
private
	int age;
	string name{};
public
	void setAge(int n) { age = n; }
	int getAge() const { return age; }
	string getName() const { return name; }
	friend ostream & operator (ostream &, const Dog);
};


ostream & operator (ostream & os, const Dog dog)
{
	os  이름    dog.getName()  ,   나이    dog.getAge()  endl;
	return os;
}
Dog[100000];
int main()
{
	Dog  p = new Dog[100000];

	default_random_engine drl;								사용할 기본 엔진 선인
	uniform_int_distributionintuid;						어떻게 분포할지 선언

	for (int i = 0; i100000 ; i++)
	{
		p[i].setAge(uid(drl));									랜덤한 int형 자료를 사용하게됨
	}

	퀵소트 (데이터, 자료형, 자료형 크기, 비교함수 포인터)
	void  로 자료형을 받을시 모든 자료형의 주소를 가져올 수 있다.

	나이로 값을 비교

	람다 람후
	[](인자)- 반환 자료형 {내용}

	qsort(p, 10000, sizeof(Dog), [](const void  d1, const void  d2) - int {
		return static_castconst Dog (d1)-getAge()
